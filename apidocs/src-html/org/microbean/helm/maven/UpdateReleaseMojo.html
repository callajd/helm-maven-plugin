057@Mojo(name = "update")
058public class UpdateReleaseMojo extends AbstractForceableMutatingReleaseMojo {
059
060
061  /*
062   * Instance fields.
063   */
064
065  
066  private final MavenProject project;
067  
068  private final MavenSession session;
069
070  @Parameter
071  private URL chartUrl;
072
073  @Parameter
074  private boolean resetValues;
075
076  @Parameter
077  private boolean reuseValues;
078  
079  @Parameter
080  private String valuesYaml;
081  
082  
083  /*
084   * Constructors.
085   */
086  
087
088  @Inject
089  public UpdateReleaseMojo(final MavenProject project, final MavenSession session) {
090    super();
091    Objects.requireNonNull(project);
092    Objects.requireNonNull(session);
093    this.project = project;
094    this.session = session;
095  }
096
097
098  /*
099   * Instance methods.
100   */
101
102  
103  @Override
104  protected void execute(final Callable<ReleaseManager> releaseManagerCallable) throws Exception {
105    Objects.requireNonNull(releaseManagerCallable);
106    final Log log = this.getLog();
107    assert log != null;
108
109    URL chartUrl = this.getChartUrl();
110    if (chartUrl == null) {
111      final Path chartDirectoryPath = Paths.get(new StringBuilder(this.project.getBuild().getDirectory()).append("/helm/").append(this.project.getArtifactId()).toString());
112      assert chartDirectoryPath != null;
113      chartUrl = chartDirectoryPath.toUri().toURL();
114      if (!Files.isDirectory(chartDirectoryPath)) {
115        throw new MojoExecutionException("Non-existent chartUrl: " + chartUrl);
116      }
117    }
118    assert chartUrl != null;
119    if (log.isDebugEnabled()) {
120      log.debug("chartUrl: " + chartUrl);
121    }
122
123    Chart.Builder chartBuilder = null;
124    try (final AbstractChartLoader<URL> chartLoader = this.createChartLoader()) {
125      if (chartLoader == null) {
126        throw new IllegalStateException("createChartLoader() == null");
127      }
128      if (log.isDebugEnabled()) {
129        log.debug("chartLoader: " + chartLoader);
130        log.debug("Loading Helm chart from " + chartUrl);
131      }
132      chartBuilder = chartLoader.load(chartUrl);
133    }
134
135    if (chartBuilder == null) {
136      throw new IllegalStateException("chartBuilder.load(\"" + chartUrl + "\") == null");
137    }
138
139    if (log.isInfoEnabled()) {
140      log.info("Loaded Helm chart from " + chartUrl);
141    }
142
143    final UpdateReleaseRequest.Builder requestBuilder = UpdateReleaseRequest.newBuilder();
144    assert requestBuilder != null;
145
146    requestBuilder.setDisableHooks(this.getDisableHooks());
147    requestBuilder.setDryRun(this.getDryRun());
148    requestBuilder.setForce(this.getForce());
149    requestBuilder.setRecreate(this.getRecreate());
150
151    final String releaseName = this.getReleaseName();
152    if (releaseName != null) {
153      requestBuilder.setName(releaseName);
154    }
155
156    requestBuilder.setResetValues(this.getResetValues());
157    requestBuilder.setReuseValues(this.getReuseValues());
158    requestBuilder.setTimeout(this.getTimeout());
159
160    final String valuesYaml = this.getValuesYaml();
161    if (valuesYaml != null && !valuesYaml.isEmpty()) {
162      final hapi.chart.ConfigOuterClass.Config.Builder values = requestBuilder.getValuesBuilder();
163      assert values != null;
164      values.setRaw(valuesYaml);
165    }
166    
167    requestBuilder.setWait(this.getWait());
168
169    final ReleaseManager releaseManager = releaseManagerCallable.call();
170    if (releaseManager == null) {
171      throw new IllegalStateException("releaseManagerCallable.call() == null");
172    }
173
174    if (log.isInfoEnabled()) {
175      log.info("Updating release " + requestBuilder.getName());
176    }
177    final Future<UpdateReleaseResponse> updateReleaseResponseFuture = releaseManager.update(requestBuilder, chartBuilder);
178    assert updateReleaseResponseFuture != null;
179    final UpdateReleaseResponse updateReleaseResponse = updateReleaseResponseFuture.get();
180    assert updateReleaseResponse != null;
181    if (log.isInfoEnabled()) {
182      final Release release = updateReleaseResponse.getRelease();
183      assert release != null;
184      log.info("Updated release " + release.getName());
185    }
186    
187  }
188
189  public URL getChartUrl() {
190    return this.chartUrl;
191  }
192
193  public void setChartUrl(final URL chartUrl) {
194    this.chartUrl = chartUrl;
195  }
196  
197  public boolean getResetValues() {
198    return this.resetValues;
199  }
200
201  public void setResetValues(final boolean resetValues) {
202    this.resetValues = resetValues;
203  }
204  
205  public boolean getReuseValues() {
206    return this.reuseValues;
207  }
208
209  public void setReuseValues(final boolean reuseValues) {
210    this.reuseValues = reuseValues;
211  }
212
213  public String getValuesYaml() {
214    return this.valuesYaml;
215  }
216
217  public void setValuesYaml(final String valuesYaml) {
218    this.valuesYaml = valuesYaml;
219  }
220
221  protected AbstractChartLoader<URL> createChartLoader() {
222    return new URLChartLoader();
223  }
