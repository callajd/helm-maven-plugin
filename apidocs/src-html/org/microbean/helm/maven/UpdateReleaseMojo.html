057/**
058 * <a
059 * href="https://docs.helm.sh/using_helm/#helm-upgrade-and-helm-rollback-upgrading-a-release-and-recovering-on-failure">Updates

061 *
062 * @author <a href="https://about.me/lairdnelson"
063 * target="_parent">Laird Nelson</a>

131  /**
132   * Creates a new {@link UpdateReleaseMojo}.
133   *
134   * @param project the {@link MavenProject} in effect; must not be
135   * {@code null}
136   *
137   * @param session the {@link MavenSession} in effect; must not be
138   * {@code null}
139   *
140   * @exception NullPointerException if either {@code project} or
141   * {@code session} is {@code null}
142   */
143  @Inject
144  public UpdateReleaseMojo(final MavenProject project, final MavenSession session) {
145    super();
146    Objects.requireNonNull(project);
147    Objects.requireNonNull(session);
148    this.project = project;
149    this.session = session;
150  }

153  /*
154   * Instance methods.
155   */
156
157
158  /**
159   * {@inheritDoc}
160   *
161   * <p>This implementation <a
162   * href="https://docs.helm.sh/using_helm/#helm-upgrade-and-helm-rollback-upgrading-a-release-and-recovering-on-failure">updates</a>
163   * the release named by the {@linkplain #getReleaseName() supplied
164   * release name} with a new <a
165   * href="https://docs.helm.sh/developing_charts/#charts">chart</a>
166   * residing at the {@linkplain #getChartUrl() indicated URL}.</p>
167   */
168  @Override
169  protected void execute(final Callable<ReleaseManager> releaseManagerCallable) throws Exception {
170    Objects.requireNonNull(releaseManagerCallable);
171    final Log log = this.getLog();
172    assert log != null;
173
174    URL chartUrl = this.getChartUrl();
175    if (chartUrl == null) {
176      final Path chartDirectoryPath = Paths.get(new StringBuilder(this.project.getBuild().getDirectory()).append("/generated-sources/helm/charts/").append(this.project.getArtifactId()).toString());
177      assert chartDirectoryPath != null;
178      chartUrl = chartDirectoryPath.toUri().toURL();
179      if (!Files.isDirectory(chartDirectoryPath)) {
180        throw new MojoExecutionException("Non-existent chartUrl: " + chartUrl);
181      }
182    }
183    assert chartUrl != null;
184    if (log.isDebugEnabled()) {
185      log.debug("chartUrl: " + chartUrl);
186    }
187
188    Chart.Builder chartBuilder = null;
189    try (final AbstractChartLoader<URL> chartLoader = this.createChartLoader()) {
190      if (chartLoader == null) {
191        throw new IllegalStateException("createChartLoader() == null");
192      }
193      if (log.isDebugEnabled()) {
194        log.debug("chartLoader: " + chartLoader);
195        log.debug("Loading Helm chart from " + chartUrl);
196      }
197      chartBuilder = chartLoader.load(chartUrl);
198    }
199
200    if (chartBuilder == null) {
201      throw new IllegalStateException("chartBuilder.load(\"" + chartUrl + "\") == null");
202    }
203
204    if (log.isInfoEnabled()) {
205      log.info("Loaded Helm chart from " + chartUrl);
206    }
207
208    final UpdateReleaseRequest.Builder requestBuilder = UpdateReleaseRequest.newBuilder();
209    assert requestBuilder != null;
210
211    requestBuilder.setDisableHooks(this.getDisableHooks());
212    requestBuilder.setDryRun(this.getDryRun());
213    requestBuilder.setForce(this.getForce());
214    requestBuilder.setRecreate(this.getRecreate());
215
216    final String releaseName = this.getReleaseName();
217    if (releaseName != null) {
218      requestBuilder.setName(releaseName);
219    }
220
221    requestBuilder.setResetValues(this.getResetValues());
222    requestBuilder.setReuseValues(this.getReuseValues());
223    requestBuilder.setTimeout(this.getTimeout());
224
225    final String valuesYaml = this.getValuesYaml();
226    if (valuesYaml != null && !valuesYaml.isEmpty()) {
227      final hapi.chart.ConfigOuterClass.Config.Builder values = requestBuilder.getValuesBuilder();
228      assert values != null;
229      values.setRaw(valuesYaml);
230    }
231    
232    requestBuilder.setWait(this.getWait());
233
234    final ReleaseManager releaseManager = releaseManagerCallable.call();
235    if (releaseManager == null) {
236      throw new IllegalStateException("releaseManagerCallable.call() == null");
237    }
238
239    if (log.isInfoEnabled()) {
240      log.info("Updating release " + requestBuilder.getName());
241    }
242    final Future<UpdateReleaseResponse> updateReleaseResponseFuture = releaseManager.update(requestBuilder, chartBuilder);
243    assert updateReleaseResponseFuture != null;
244    final UpdateReleaseResponse updateReleaseResponse = updateReleaseResponseFuture.get();
245    assert updateReleaseResponse != null;
246    if (log.isInfoEnabled()) {
247      final Release release = updateReleaseResponse.getRelease();
248      assert release != null;
249      log.info("Updated release " + release.getName());
250    }
251    
252  }

254  /**
255   * Returns a {@link URL} identifying a Helm chart that can be read
256   * by the {@link AbstractChartLoader} produced by the {@link
257   * #createChartLoader()} method.
258   *
259   * <p>This method may return {@code null}.</p>
260   *
261   * @return a {@link URL} to a Helm chart, or {@code null}
262   *
263   * @see #setChartUrl(URL)
264   */
265  public URL getChartUrl() {
266    return this.chartUrl;
267  }
268  
269  /**
270   * Sets the {@link URL} identifying a Helm chart that can be read
271   * by the {@link AbstractChartLoader} produced by the {@link
272   * #createChartLoader()} method.
273   *
274   * @param chartUrl the {@link URL} identifying a Helm chart that can
275   * be read by the {@link AbstractChartLoader} produced by the {@link
276   * #createChartLoader()} method; may be {@code null}
277   */
278  public void setChartUrl(final URL chartUrl) {
279    this.chartUrl = chartUrl;
280  }
281
282  /**
283   * Returns {@code true} if, during the update, values should be
284   * reset to the values built in to the {@linkplain #getChartUrl()
285   * new chart}.
286   *
287   * @return {@code true} if, during the update, values should be
288   * reset to the values built in to the {@linkplain #getChartUrl()
289   * new chart}; {@code false} otherwise
290   *
291   * @see #setResetValues(boolean)
292   */
293  public boolean getResetValues() {
294    return this.resetValues;
295  }
296
297  /**
298   * Sets whether, during the update, values should be
299   * reset to the values built in to the {@linkplain #getChartUrl()
300   * new chart}.
301   *
302   * @param resetValues if {@code true}, during the update values will
303   * be reset to the values built in to the {@linkplain #getChartUrl()
304   * new chart}
305   *
306   * @see #getResetValues()
307   */
308  public void setResetValues(final boolean resetValues) {
309    this.resetValues = resetValues;
310  }
311
312  /**
313   * Returns {@code true} if, during the update, any new values should
314   * be merged with those present in the prior version of the release
315   * being updated.
316   *
317   * @return {@code true} if, during the update, any new values should
318   * be merged with those present in the prior version of the release
319   * being updated; {@code false} otherwise
320   *
321   * @see #setReuseValues(boolean)
322   */
323  public boolean getReuseValues() {
324    return this.reuseValues;
325  }
326
327  /**
328   * Sets whether, during the update, any new values should
329   * be merged with those present in the prior version of the release
330   * being updated.
331   *
332   * @param reuseValues if {@code true} during the update any new
333   * values will be merged with those present in the prior version of
334   * the release being updated
335   *
336   * @see #getReuseValues()
337   */
338  public void setReuseValues(final boolean reuseValues) {
339    this.reuseValues = reuseValues;
340  }
341
342  /**
343   * Returns a YAML {@link String} representing the values to use to
344   * customize the update.
345   *
346   * <p>This method may return {@code null}.</p>
347   *
348   * <p>Overrides of this method may return {@code null}.</p>
349   *
350   * @return a YAML {@link String} representing the values to use to
351   * customize the update, or {@code null}
352   *
353   * @see #setValuesYaml(String)
354   */
355  public String getValuesYaml() {
356    return this.valuesYaml;
357  }
358
359  /**
360   * Installs a YAML {@link String} representing the values to use to
361   * customize the update.
362   *
363   * @param valuesYaml the YAML {@link String} representing the values to use to
364   * customize the update; may be {@code null}
365   *
366   * @see #getValuesYaml()
367   */
368  public void setValuesYaml(final String valuesYaml) {
369    this.valuesYaml = valuesYaml;
370  }
371
372  /**
373   * Creates and returns an {@link AbstractChartLoader} capable of
374   * loading a Helm chart from a {@link URL}.
375   *
376   * <p>This method never returns {@code null}.</p>
377   *
378   * <p>Overrides of this method must not return {@code null}.</p>
379   *
380   * <p>This implementation returns a new {@link URLChartLoader}.</p>
381   *
382   * @return a new {@link AbstractChartLoader} implementation; never
383   * {@code null}
384   */
385  protected AbstractChartLoader<URL> createChartLoader() {
386    return new URLChartLoader();
387  }
