043@Mojo(name = "list")
044public class ListReleasesMojo extends AbstractReleaseMojo {
045
046  @Parameter
047  private String filter;
048
049  @Parameter(defaultValue = "256")
050  private long limit;
051  
052  @Parameter
053  private String namespace;
054
055  @Parameter
056  private String offset;
057
058  @Parameter
059  private SortBy sortBy;
060
061  @Parameter
062  private SortOrder sortOrder;
063
064  @Parameter
065  private List<Status.Code> statusCodes;
066
067  @Parameter(alias = "releaseDiscoveryListenersList")
068  private List<ReleaseDiscoveryListener> releaseDiscoveryListeners;
069  
070
071  /*
072   * Constructors.
073   */
074  
075  
076  public ListReleasesMojo() {
077    super();
078  }
079
080
081  /*
082   * Protected instance methods.
083   */
084  
085
086  @Override
087  protected void execute(final Callable<ReleaseManager> releaseManagerCallable) throws Exception {
088    Objects.requireNonNull(releaseManagerCallable);
089    final Log log = this.getLog();
090    assert log != null;
091
092    final Collection<? extends ReleaseDiscoveryListener> listeners = this.getReleaseDiscoveryListenersList();
093    if (listeners == null || listeners.isEmpty()) {
094      if (log.isInfoEnabled()) {
095        log.info("Skipping execution because there are no ReleaseDiscoveryListeners specified.");
096      }
097      return;
098    }
099
100    final ListReleasesRequest.Builder requestBuilder = ListReleasesRequest.newBuilder();
101    assert requestBuilder != null;
102
103    final String filter = this.getFilter();
104    if (filter != null) {
105      requestBuilder.setFilter(filter);
106    }
107
108    requestBuilder.setLimit(this.getLimit());
109
110    String namespace = this.getNamespace();
111    if (namespace == null || namespace.isEmpty()) {
112      final io.fabric8.kubernetes.client.Config configuration = this.getClientConfiguration();
113      if (configuration == null) {
114        namespace = "default";
115      } else {
116        namespace = configuration.getNamespace();
117        if (namespace == null || namespace.isEmpty()) {
118          namespace = "default";
119        }
120      }
121    }
122    this.validateNamespace(namespace);
123    requestBuilder.setNamespace(namespace);
124
125    final String offset = this.getOffset();
126    if (offset != null) {
127      requestBuilder.setOffset(offset);
128    }
129
130    final SortBy sortBy = this.getSortBy();
131    if (sortBy != null) {
132      requestBuilder.setSortBy(sortBy);
133    }
134
135    final SortOrder sortOrder = this.getSortOrder();
136    if (sortOrder != null) {
137      requestBuilder.setSortOrder(sortOrder);
138    }
139
140    final Iterable<Status.Code> statusCodes = this.getStatusCodes();
141    if (statusCodes != null) {
142      requestBuilder.addAllStatusCodes(statusCodes);
143    }
144
145    final ReleaseManager releaseManager = releaseManagerCallable.call();
146    if (releaseManager == null) {
147      throw new IllegalStateException("releaseManagerCallable.call() == null");
148    }
149
150    if (log.isInfoEnabled()) {
151      log.info("Listing releases in namespace " + namespace);
152    }
153    
154    final Iterator<? extends ListReleasesResponse> listReleasesResponseIterator = releaseManager.list(requestBuilder.build());
155    assert listReleasesResponseIterator != null;
156    while (listReleasesResponseIterator.hasNext()) {
157      final ListReleasesResponse response = listReleasesResponseIterator.next();
158      assert response != null;
159      final ReleaseDiscoveryEvent event = new ReleaseDiscoveryEvent(this, log, response);
160      for (final ReleaseDiscoveryListener listener : listeners) {
161        if (listener != null) {
162          listener.releaseDiscovered(event);
163        }
164      }
165    }
166    
167  }
168

185  }

189  }
190

192    return this.namespace;
193  }
194

196    this.validateNamespace(namespace);
197    this.namespace = namespace;
198  }
199

201    return this.offset;
202  }
203

205    this.offset = offset;
206  }

208  public SortBy getSortBy() {
209    return this.sortBy;
210  }
211
212  public void setSortBy(final SortBy sortBy) {
213    this.sortBy = sortBy;
214  }
215
216  public SortOrder getSortOrder() {
217    return this.sortOrder;
218  }
219
220  public void setSortOrder(final SortOrder sortOrder) {
221    this.sortOrder = sortOrder;
222  }

224  public List<Status.Code> getStatusCodes() {
225    return this.statusCodes;
226  }
227
228  public void setStatusCodes(final List<Status.Code> statusCodes) {
229    this.statusCodes = statusCodes;
230  }
231
232  public void addReleaseDiscoveryListener(final ReleaseDiscoveryListener listener) {
233    if (listener != null) {
234      if (this.releaseDiscoveryListeners == null) {
235        this.releaseDiscoveryListeners = new ArrayList<>();      
236      }
237      this.releaseDiscoveryListeners.add(listener);
238    }
239  }
240
241  public void removeReleaseDiscoveryListener(final ReleaseDiscoveryListener listener) {
242    if (listener != null && this.releaseDiscoveryListeners != null) {
243      this.releaseDiscoveryListeners.remove(listener);
244    }
245  }
246  
247  public ReleaseDiscoveryListener[] getReleaseDiscoveryListeners() {
248    final Collection<ReleaseDiscoveryListener> listeners = this.getReleaseDiscoveryListenersList();
249    if (listeners == null || listeners.isEmpty()) {
250      return new ReleaseDiscoveryListener[0];
251    } else {
252      return listeners.toArray(new ReleaseDiscoveryListener[listeners.size()]);
253    }
254  }
255
256  public List<ReleaseDiscoveryListener> getReleaseDiscoveryListenersList() {
257    return this.releaseDiscoveryListeners;
258  }
259
260  public void setReleaseDiscoveryListenersList(final List<ReleaseDiscoveryListener> releaseDiscoveryListeners) {
261    this.releaseDiscoveryListeners = releaseDiscoveryListeners;
262  }
