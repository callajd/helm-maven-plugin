017package org.microbean.helm.maven;
018
019import java.util.Objects;
020
021import java.util.concurrent.Callable;
022import java.util.concurrent.Future;
023
024import hapi.release.ReleaseOuterClass.Release;
025
026import hapi.services.tiller.Tiller.RollbackReleaseRequest;
027import hapi.services.tiller.Tiller.RollbackReleaseResponse;
028
029import org.apache.maven.plugin.logging.Log;
030
031import org.apache.maven.plugins.annotations.Mojo;
032import org.apache.maven.plugins.annotations.Parameter;
033
034import org.microbean.helm.ReleaseManager;
035
036@Mojo(name = "rollback")
037public class RollbackReleaseMojo extends AbstractForceableMutatingReleaseMojo {
038
039
040  /*
041   * Instance fields.
042   */
043
044  
045  @Parameter
046  private int version;
047
048
049  /*
050   * Constructors.
051   */
052  
053
054  public RollbackReleaseMojo() {
055    super();
056  }
057
058
059  /*
060   * Instance methods.
061   */
062
063  
064  @Override
065  protected void execute(final Callable<ReleaseManager> releaseManagerCallable) throws Exception {
066    Objects.requireNonNull(releaseManagerCallable);
067    final Log log = this.getLog();
068    assert log != null;
069
070    final RollbackReleaseRequest.Builder requestBuilder = RollbackReleaseRequest.newBuilder();
071    assert requestBuilder != null;
072
073    requestBuilder.setDisableHooks(this.getDisableHooks());
074    requestBuilder.setDryRun(this.getDryRun());
075    requestBuilder.setForce(this.getForce());
076    requestBuilder.setRecreate(this.getRecreate());
077
078    final String releaseName = this.getReleaseName();
079    if (releaseName != null) {
080      requestBuilder.setName(releaseName);
081    }
082    
083    requestBuilder.setTimeout(this.getTimeout());
084    requestBuilder.setVersion(this.getVersion());
085    requestBuilder.setWait(this.getWait());
086
087    final ReleaseManager releaseManager = releaseManagerCallable.call();
088    if (releaseManager == null) {
089      throw new IllegalStateException("releaseManagerCallable.call() == null");
090    }
091
092    if (log.isInfoEnabled()) {
093      log.info("Rolling back release " + requestBuilder.getName());
094    }
095    final Future<RollbackReleaseResponse> rollbackReleaseResponseFuture = releaseManager.rollback(requestBuilder.build());
096    assert rollbackReleaseResponseFuture != null;
097    final RollbackReleaseResponse rollbackReleaseResponse = rollbackReleaseResponseFuture.get();
098    assert rollbackReleaseResponse != null;
099    if (log.isInfoEnabled()) {
100      final Release release = rollbackReleaseResponse.getRelease();
101      assert release != null;
102      log.info("Rolled back release " + release.getName());
103    }
104    
105  }
106
107  public int getVersion() {
108    return this.version;
109  }
110
111  public void setVersion(final int version) {
112    this.version = version;
113  }
114
115}
